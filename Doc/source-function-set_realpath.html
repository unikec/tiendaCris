63:         // Security check to make sure the path is NOT a URL. No remote file inclusion!
64:         if (preg_match('#^(http:\/\/|https:\/\/|www\.|ftp|php:\/\/)#i', $path) OR filter_var($path, FILTER_VALIDATE_IP) === $path)

76:             show_error('Not a valid path: '.$path);
