 53:     /**
 54:      * Number of rows in the result set
 55:      *
 56:      * @return  int
 57:      */
 58:     public function num_rows()
 59:     {
 60:         if (is_int($this->num_rows))
 61:         {
 62:             return $this->num_rows;
 63:         }
 64:         elseif (count($this->result_array) > 0)
 65:         {
 66:             return $this->num_rows = count($this->result_array);
 67:         }
 68:         elseif (count($this->result_object) > 0)
 69:         {
 70:             return $this->num_rows = count($this->result_object);
 71:         }
 72:         elseif (($num_rows = $this->result_id->rowCount()) > 0)
 73:         {
 74:             return $this->num_rows = $num_rows;
 75:         }
 76: 
 77:         return $this->num_rows = count($this->result_array());
 78:     }
 79:

 81: 
 82:     /**
 83:      * Number of fields in the result set
 84:      *
 85:      * @return  int
 86:      */
 87:     public function num_fields()
 88:     {
 89:         return $this->result_id->columnCount();
 90:     }

 92:     // --------------------------------------------------------------------

 94:     /**
 95:      * Fetch Field Names
 96:      *
 97:      * Generates an array of column names
 98:      *
 99:      * @return  bool
100:      */
101:     public function list_fields()
102:     {
103:         $field_names = array();
104:         for ($i = 0, $c = $this->num_fields(); $i < $c; $i++)
105:         {
106:             // Might trigger an E_WARNING due to not all subdrivers
107:             // supporting getColumnMeta()
108:             $field_names[$i] = @$this->result_id->getColumnMeta($i);
109:             $field_names[$i] = $field_names[$i]['name'];
110:         }
111: 
112:         return $field_names;
113:     }
114: 
115:     // --------------------------------------------------------------------
116: 
117:     /**
118:      * Field data
119:      *
120:      * Generates an array of objects containing field meta-data
121:      *
122:      * @return  array
123:      */
124:     public function field_data()
125:     {
126:         try
127:         {
128:             $retval = array();
129: 
130:             for ($i = 0, $c = $this->num_fields(); $i < $c; $i++)
131:             {
132:                 $field = $this->result_id->getColumnMeta($i);
133: 
134:                 $retval[$i]         = new stdClass();
135:                 $retval[$i]->name       = $field['name'];
136:                 $retval[$i]->type       = $field['native_type'];
137:                 $retval[$i]->max_length     = ($field['len'] > 0) ? $field['len'] : NULL;
138:                 $retval[$i]->primary_key    = (int) ( ! empty($field['flags']) && in_array('primary_key', $field['flags'], TRUE));
139:             }
140: 
141:             return $retval;
142:         }
143:         catch (Exception $e)
144:         {
145:             if ($this->db->db_debug)
146:             {
147:                 return $this->db->display_error('db_unsupported_feature');
148:             }
149: 
150:             return FALSE;
151:         }
152:     }
