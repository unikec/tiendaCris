297:     /**
298:      * creates a file at given path
299:      *
300:      * @param   string  $path     the path to open
301:      * @param   string  $mode     mode for opening

306:     {
307:         $names = $this->splitPath($path);
308:         if (empty($names['dirname']) === true) {
309:             if (($options & STREAM_REPORT_ERRORS) === STREAM_REPORT_ERRORS) {
310:                 trigger_error('File ' . $names['basename'] . ' does not exist', E_USER_WARNING);
311:             }
312: 
313:             return false;
314:         }
315: 
316:         $dir = $this->getContentOfType($names['dirname'], vfsStreamContent::TYPE_DIR);
317:         if (null === $dir) {
318:             if (($options & STREAM_REPORT_ERRORS) === STREAM_REPORT_ERRORS) {
319:                 trigger_error('Directory ' . $names['dirname'] . ' does not exist', E_USER_WARNING);
320:             }
321: 
322:             return false;
323:         } elseif ($dir->hasChild($names['basename']) === true) {
324:             if (($options & STREAM_REPORT_ERRORS) === STREAM_REPORT_ERRORS) {
325:                 trigger_error('Directory ' . $names['dirname'] . ' already contains a director named ' . $names['basename'], E_USER_WARNING);
326:             }

328:             return false;
329:         }

332:             if (($options & STREAM_REPORT_ERRORS) === STREAM_REPORT_ERRORS) {
333:                 trigger_error('Can not open non-existing file ' . $path . ' for reading', E_USER_WARNING);

336:             return false;
337:         }

341:                 trigger_error('Can not create new file in non-writable path ' . $names['dirname'], E_USER_WARNING);

344:             return false;
